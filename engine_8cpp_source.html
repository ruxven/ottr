21 out = Aggregation{};
22
23 // Build list of included dates in order
24 std::vector<const Day*> included_days;
25 included_days.reserve(world.days.size());
26 for (const auto& day : world.days) {

28 included_days.push_back(&day);

31 std::sort(included_days.begin(), included_days.end(), [](const Day* a, const Day* b){ return a->date.ord < b->date.ord; });
32 out.dates_in_order.reserve(included_days.size());
33 for (auto* d : included_days) out.dates_in_order.push_back(d->date);
34
35 RunningTotals rt;
36
37 for (const Day* day : included_days) {
38 const auto& logs = day->logs;
39 if (logs.size() < 2) continue; // validation should have caught
40 for (size_t i = 0; i + 1 < logs.size(); ++i) {
41 const Log& start = logs[i];
42 const Log& end = logs[i+1];
43 Tick slice_ticks = end.time_ticks - start.time_ticks;
44 if (slice_ticks <= 0) continue;
45 if (start.kind == LogKind::None) {
46 continue; // uncharged slice
47 }
48 if (start.kind == LogKind::Charge) {
49 add_allocation(out, day->date, start.ref_id, slice_ticks);
50 rt.charged[start.ref_id] += slice_ticks;
51 continue;
52 }
53 // Task-based or possibly a direct-charge disguised as task name
54 // If ref id is a known charge, treat as direct charge
55 auto charge_it = world.charges.find(start.ref_id);
56 if (charge_it != world.charges.end()) {
57 add_allocation(out, day->date, start.ref_id, slice_ticks);
58 rt.charged[start.ref_id] += slice_ticks;
59 continue;
60 }
61 auto task_it = world.tasks.find(start.ref_id);
62 if (task_it == world.tasks.end()) {
63 // Unknown task at runtime; validation should catch. Skip.
64 continue;
65 }
66 const Task& task = task_it->second;
67 // If no weights -> uncharged
68 if (task.weights.empty()) continue;
69
70 // Build allocation request
71 AllocationRequest req;
72 req.slice_ticks = slice_ticks;
73 req.entries.reserve(task.weights.size());
74 for (const auto& w : task.weights) {
75 AllocationEntry e;
76 e.charge_id = w.first;
77 e.weight = w.second;
78 auto cit = world.charges.find(e.charge_id);
79 if (cit != world.charges.end()) {
80 e.priority = cit->second.priority;
81 e.remaining_budget = cit->second.budget_ticks - (rt.charged[e.charge_id]);
82 } else {
83 // Unknown charge in weights -> skip; validator should catch
84 continue;
85 }
86 // Record fractional info will be computed by allocator
87 req.entries.push_back(std::move(e));
88 }
89 auto result = allocate_weighted(req);
90 for (const auto& a : result) {
91 if (a.ticks > 0) {
92 add_allocation(out, day->date, a.charge_id, a.ticks);
93 rt.charged[a.charge_id] += a.ticks;
94 }
95 }
96 }
97 }
