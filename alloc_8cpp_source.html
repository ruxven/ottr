13std::vector<AllocationResultEntry> allocate_weighted(const AllocationRequest& req) {
14 std::vector<AllocationResultEntry> out;
15 if (req.slice_ticks <= 0 || req.entries.empty()) return out;
16
17 long long denom = 0;
18 for (const auto& e : req.entries) {
19 if (e.weight > 0) denom += e.weight;
20 }
21 if (denom <= 0) return out; // uncharged per spec
22
23 std::vector<WorkItem> work;
24 work.reserve(req.entries.size());
25 long long base_sum = 0;
26 for (size_t i = 0; i < req.entries.size(); ++i) {
27 const auto& e = req.entries[i];
28 if (e.weight <= 0) continue;
29 long long ideal_numer = static_cast<long long>(req.slice_ticks) * static_cast<long long>(e.weight);
30 long long base = ideal_numer / denom; // integer division
31 long long frac = ideal_numer % denom;
32 work.push_back(WorkItem{i, base, frac});
33 base_sum += base;
34 }
35
36 long long remainder = static_cast<long long>(req.slice_ticks) - base_sum;
37 if (remainder < 0) remainder = 0; // safety
38
39 auto choose_idx = [&](const std::vector<WorkItem>& items) -> size_t {
40 size_t best = items[0].idx;
41 // We'll scan req.entries via indices considering the comparator
42 for (const auto& wi : items) {
43 size_t j = wi.idx;
44 size_t b = best;
45 const auto& ej = req.entries[j];
46 const auto& eb = req.entries[b];
47 // 1) Higher priority first
48 if (ej.priority != eb.priority) {
49 if (ej.priority > eb.priority) best = j; // higher wins
50 continue;
51 }
52 // 2) Larger remaining budget
53 if (ej.remaining_budget != eb.remaining_budget) {
54 if (ej.remaining_budget > eb.remaining_budget) best = j;
55 continue;
56 }
57 // 3) Larger fractional remainder (frac)
58 long long frac_b = 0;
59 long long frac_j = 0;
60 for (const auto& w : work) {
61 if (w.idx == b) frac_b = w.frac;
62 if (w.idx == j) frac_j = w.frac;
63 }
64 if (frac_j != frac_b) {
65 if (frac_j > frac_b) best = j;
66 continue;
67 }
68 // 4) Lexicographically smaller charge id
69 if (ej.charge_id < eb.charge_id) best = j;
70 }
71 return best;
72 };
73
74 // Prepare result with base allocations
75 out.resize(req.entries.size());
76 for (size_t i = 0; i < req.entries.size(); ++i) {
77 out[i].charge_id = req.entries[i].charge_id;
78 out[i].ticks = 0;
79 }
80 for (const auto& w : work) {
81 out[w.idx].ticks += static_cast<Tick>(w.base);
82 }
83
84 // Distribute remainder one tick at a time
85 while (remainder > 0 && !work.empty()) {
86 size_t winner_idx = choose_idx(work);
87 // increment the base of the winner for next comparison steps
88 for (auto& w : work) {
89 if (w.idx == winner_idx) { w.base += 1; break; }
90 }
91 out[winner_idx].ticks += 1;
92 remainder -= 1;
93 }
94
95 return out;
96}
